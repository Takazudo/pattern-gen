# API Reference

## Core Functions

### `renderPattern(options)`

Render a pattern to a PNG buffer (Node.js only).

```typescript
import { renderPattern } from '@takazudo/pattern-gen';

const result = await renderPattern({
  slug: 'my-pattern',    // Text seed
  type: 'voronoi',       // Pattern type name
  size: 800,             // Canvas size (px)
  zoom: 1,               // Zoom factor
  colorScheme: 'nord',   // Color scheme name or "random"
  bg: '#1a1a2e',         // Override background (optional)
});

// result.buffer: Buffer (PNG)
// result.patternName: string
// result.colorSchemeName: string
// result.width: number
// result.height: number
```

### `renderPatternToCanvas(ctx, slug, type, options?)`

Render a pattern on a browser Canvas 2D context.

```typescript
import { renderPatternToCanvas } from '@takazudo/pattern-gen';

const canvas = document.querySelector('canvas');
const ctx = canvas.getContext('2d');

const { colorSchemeName } = renderPatternToCanvas(
  ctx,
  'my-pattern',
  'voronoi',
  { zoom: 2, colorScheme: 'dracula' }
);
```

### `renderOgpFromConfig(config)`

Render an OGP image (1200×630) from a serialized config (Node.js only).

```typescript
import { renderOgpFromConfig, parseOgpConfig } from '@takazudo/pattern-gen';

const json = fs.readFileSync('ogp-config.json', 'utf-8');
const config = parseOgpConfig(json);
const result = await renderOgpFromConfig(config);

// result.buffer: Buffer (1200×630 PNG)
fs.writeFileSync('ogp-image.png', result.buffer);
```

### `renderOgpEditorFromConfig(config)`

Render an OGP image from an editor config with layers and frames (Node.js only).

```typescript
import { renderOgpEditorFromConfig, parseOgpEditorConfig } from '@takazudo/pattern-gen';

const json = fs.readFileSync('editor-config.json', 'utf-8');
const config = parseOgpEditorConfig(json);
const result = await renderOgpEditorFromConfig(config);

// result.buffer: Buffer (1200×630 PNG with background + layers + frame)
fs.writeFileSync('ogp-image.png', result.buffer);
```

## OGP Config

### `OgpConfig` Type

```typescript
import type { OgpConfig, OgpCropRegion } from '@takazudo/pattern-gen';

interface OgpConfig {
  version: 1;
  slug: string;              // Text seed
  type: string;              // Pattern type name
  colorScheme: string;       // Color scheme name
  zoom: number;              // Zoom factor
  translateX: number;        // Horizontal offset (-1 to 1)
  translateY: number;        // Vertical offset (-1 to 1)
  useTranslate: boolean;     // Enable panning mode
  params: Record<string, number>;  // Pattern parameters
  hsl: { h: number; s: number; l: number };  // HSL adjustments
  crop: OgpCropRegion;       // Normalized crop region (0-1 fractions)
}

interface OgpCropRegion {
  x: number;      // Left edge (0-1)
  y: number;      // Top edge (0-1)
  width: number;  // Width (0-1)
  height: number; // Height (0-1)
}
```

### `serializeOgpConfig(config)`

Serialize an OgpConfig to pretty-printed JSON.

```typescript
import { serializeOgpConfig } from '@takazudo/pattern-gen';

const json = serializeOgpConfig(config);
// Pretty-printed JSON string
```

### `parseOgpConfig(json)`

Parse and validate a JSON string into an OgpConfig. Throws on invalid input.

```typescript
import { parseOgpConfig } from '@takazudo/pattern-gen';

const config = parseOgpConfig(jsonString);
// Validates: version, types, ranges, crop bounds
// Throws descriptive errors on invalid input
```

## Utility Functions

### `hashString(str)`

FNV-1a hash: string to 32-bit unsigned integer.

```typescript
import { hashString } from '@takazudo/pattern-gen';
const seed = hashString('hello-world'); // deterministic number
```

### `createRandom(seed)`

Mulberry32 PRNG: seed to deterministic random number generator.

```typescript
import { createRandom } from '@takazudo/pattern-gen';
const rand = createRandom(42);
rand(); // 0.xxx (always same sequence for same seed)
```

### `createNoise2D(rand)`

2D simplex noise from seeded PRNG. Returns `(x, y) => [-1, 1]`.

```typescript
import { createRandom, createNoise2D } from '@takazudo/pattern-gen';
const noise = createNoise2D(createRandom(42));
noise(1.5, 2.3); // deterministic float in [-1, 1]
```

### `fbm(noise, x, y, octaves?, lacunarity?, gain?)`

Fractal Brownian Motion — sum multiple octaves of noise.

## Frame System

### `FrameGenerator` Type

```typescript
import type { FrameGenerator, FrameParamDef, FrameRenderOptions } from '@takazudo/pattern-gen';

interface FrameGenerator {
  name: string;           // URL-safe slug (e.g., 'simple-line')
  displayName: string;    // Human-readable name
  description: string;
  paramDefs: FrameParamDef[];
  render(ctx: CanvasRenderingContext2D, options: FrameRenderOptions, params: Record<string, number | string>): void;
}

interface FrameRenderOptions {
  width: number;          // Canvas width (1200 for OGP)
  height: number;         // Canvas height (630 for OGP)
  rand: () => number;     // Seeded PRNG for deterministic frames
}
```

### Frame Parameter Types

```typescript
type FrameParamDef = SliderFrameParam | ColorFrameParam | SelectFrameParam | ToggleFrameParam;

// slider: numeric range (e.g., border width 2-40)
// color: hex string with alpha (#RRGGBB or #RRGGBBAA)
// select: enumerated options (e.g., line style: solid/dashed)
// toggle: boolean (0 or 1)
```

```typescript
import type { OgpEditorConfig, FrameConfig } from '@takazudo/pattern-gen';

interface OgpEditorConfig {
```

## Adding a Custom Pattern

```typescript
import type { PatternGenerator, PatternOptions } from '@takazudo/pattern-gen';

const myPattern: PatternGenerator = {
  name: 'my-pattern',
  displayName: 'My Pattern',
  description: 'A custom pattern',

  generate(ctx: CanvasRenderingContext2D, options: PatternOptions): void {
    const { width, height, rand, colorScheme, zoom } = options;
    const bg = colorScheme.palette[0];      // Background
    const fgColors = colorScheme.palette.slice(1); // 7 foreground colors

    ctx.fillStyle = bg;
    ctx.fillRect(0, 0, width, height);

    // Use rand() for ALL randomness (deterministic)
    // Draw your pattern on ctx...
  },
};
```
